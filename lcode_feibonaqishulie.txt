class Solution {
public:
    int fib(int n) {
        if(n<=3)
        {
            int *dp=new int[3];
            dp[0]=0;
            dp[1]=1;
            dp[2]=1;
            return dp[n];
        }
            int  *dp=new int[n];
            dp[0]=0;
            dp[1]=1;
            dp[2]=1;
            for(int i=3;i<=n;i++)
            dp[i]=(dp[i-1]+dp[i-2])%1000000007;
            return dp[n];
        
    }
};
